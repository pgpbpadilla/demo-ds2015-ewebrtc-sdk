//Step 1. Get Phone Objects

function onError(data) {
    console.log(data.error);
}

function configurePhone() {
    var phone = ATT.rtc.configure(getPhone);
}      
    
function getPhone() {
    phone = ATT.rtc.Phone.getPhone();
    registerEvents (); 
}

function registerEvents() {
    phone.on('error', onError); 
}
